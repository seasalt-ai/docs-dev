Cart Config Example¶

MENU_PATHS:
  - './data/menu.json'

# Thresholds for matching/suggesting with fuzzywuzzy and elasticsearch
FUZZY_MATCH_THRESH: 85  # constant for fuzzy matching threshold
FUZZY_SUGGEST_THRESH: 50  # constant for fuzzy matching similar items
ES_MATCH_THRESH: 3.5  # constant for ItemSearch matching threshold
ES_SUGGEST_THRESH: 1.8  # constant for ItemSearch matching similar items
ES_UNIQUE_THRESH: 0.3 # a constant for determining if a top result is unique enough to be an exact match

# categories of attributive options that should precede the item name
ATTRIBUTIVE_OPTIONS:
  - size

# host and port info for elasticsearch service
ES_TRY_LOCAL_FIRST: True
# On the k8s cluster, login/secret will be set automatically.
# ES_PORT: 9200
# ES_HOST: elastic.seasalt.ai
# ES_LOGIN: elastic
# ES_SECRET: J248bGk1rg7zOhl5Tq404ut6
ES_MULTI_MATCH_TYPE: cross_fields  # by default it's best_fields
ES_MULTI_MATCH_FIELDS: ["aliases", "modifier_names", "tags", "categories", "name", "ingredients", "description"]
