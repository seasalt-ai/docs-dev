Dialogue.yml Example¶

$[greet]:
  rank_score: 100
  conditions:
    - INTENT.name == 'greet'
  actions:
    - utter_greet
    - action_listen

$[display menu]:
  conditions:
    - INTENT.name == 'display_menu'
  actions:
    - action_display_menu
    - action_listen

$[select item]:
  conditions:
    - INTENT.name == 'select_item'
  actions:
    - action_validate_item
  connections:
    - $[add item]:
        direct_connection: true
        conditions:
          - SLOTS.item is not None
        actions:
          - action_add_item
          - action_display_cart
          - action_listen
    - $[add item no item]:
        direct_connection: true
        conditions:
          - SLOTS.item is None
        actions:
          - utter_no_item_selected
          - action_display_menu
          - action_listen

$[faq]:
  rank_score: 100
  conditions:
    - INTENT.name == 'faq'
  actions:
    - utter_faq
    - action_listen
