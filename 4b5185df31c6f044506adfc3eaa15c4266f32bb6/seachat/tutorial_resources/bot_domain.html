Domain Example¶

intents:
  - greet
  - faq
  - display_menu
  - select_item

entities:
  - item

slots:
  item:
    type: text

responses:
  utter_default:
  - text: I'm sorry, I didn't understand that. 

  utter_display_menu:
  - text: Here's what's on the menu; Americano, House Coffee, Decaf Coffee, Breakfast Combo, Muffin, Croissant, Bagel, and Panini.

  utter_select_item:
  - text: Okay, you want to select an item.

  utter_added_item:
  - text: Okay, I'll add a {item} to you cart.

  utter_no_item_selected:
  - text: You didn't specify what item you want.

  utter_greet:
  - text: "Hey! What can I get you?"

actions:
  - utter_faq
  - action_session_start
  - action_display_menu
  - action_validate_item
  - action_add_item
  - action_display_cart

session_config:
  session_expiration_time: 60
  carry_over_slots_to_new_session: true
