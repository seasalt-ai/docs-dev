Config Example

# Configuration for Rasa NLU.
# https://rasa.com/docs/rasa/nlu/components/
language: en

importers:
  - name: seachat_plugins.importers.faq_importer.FAQImporter
    faq_paths:
      - data/faq.csv
    augment_with_synonyms: false
    timezone: America/Los_Angeles
  - name: RasaFileImporter

pipeline:
  - name: WhitespaceTokenizer
  - name: RegexFeaturizer
  - name: LexicalSyntacticFeaturizer
  - name: CountVectorsFeaturizer
  - name: CountVectorsFeaturizer
    analyzer: "char_wb"
    min_ngram: 1
    max_ngram: 4
  - name: DIETClassifier
    epochs: 100
  - name: EntitySynonymMapper
  - name: seachat_plugins.nlu.selectors.elastic_selector.ElasticSelector
    retrieval_intent: faq
    batch_size: 2048
    method: text
    es_indexing:
      analyzer: standard
      index_name: tutorial_faq
      # On the k8s cluster, login/secret will be set automatically.
      host: localhost
      port: 9200
    bert_client:
      enabled: false
      host: bert-as-service
      port: 5555
      port_out: 5556
    eval:
      enabled: false
      metric: map
      N: 1
    reranker:
      enabled: false

# Configuration for Rasa Core.
# https://rasa.com/docs/rasa/core/policies/
policies:
  - name: seachat_plugins.core.policy.ngchat.ngChatPolicy
    nlu_threshold: .20
    visual:
      visualization_directory: ./visual
      output_static_visualization: false
      vega_spec_path: ./visual/flow_data.json
      output_at_each_turn: false
      output_live_update: false
      output_vega: false
      output_dot: false
